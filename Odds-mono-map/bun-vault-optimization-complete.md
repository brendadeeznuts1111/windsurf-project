// Intelligent filename conversion
class BunNamingStandardsFixer {
    private convertToKebabCase(filename: string): string {
        // Converts UPPER_CASE, PascalCase, snake_case to kebab-case
        return filename
            .replace(/_/g, '-')
            .replace(/([a-z])([A-Z])/g, '$1-$2')
            .toLowerCase();
    }
}
